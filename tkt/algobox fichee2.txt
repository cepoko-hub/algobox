exercice 1:

VARIABLES
  i EST_DU_TYPE NOMBRE
  y EST_DU_TYPE NOMBRE
  ligne EST_DU_TYPE CHAINE
DEBUT_ALGORITHME
  AFFICHER* ligne
  FIN_POUR
  AFFICHER "combien de ligne fait la pyramide ?"
  LIRE ligne
  POUR ligne ALLANT_DE 1 A i
  ligne PREND_LA_VALEUR ""
  POUR y ALLANT_DE 1 A i
  ligne PREND_LA_VALEUR ligne+"⬛"
  FIN_POUR
FIN_ALGORITHME

  i EST_DU_TYPE NOMBRE
  y EST_DU_TYPE NOMBRE
  ligne EST_DU_TYPE NOMBRE

  AFFICHER* ligne
  FIN_POUR
  AFFICHER "entré la taille du damier"
  LIRE ligne

  ligne PREND_LA_VALEUR ""

  FIN_POUR
